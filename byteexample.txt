class ByteExample{
	public static void main(String args[])
	{
		byte age;
		byte month;
		byte weekdays;
		byte hoursinaday;
		byte wetherindegrees;
		
		age = 60;
		month = 12;
		weekdays = 7;
		hoursinaday = 24;

		System.out.println("Age is"+age);
		System.out.println("number of months in a year"+month);
		System.out.println("number of days in week"+weekdays);
		System.out.println("number of hours in a day"+hoursinaday);
		System.out.println("wether in srinagar "+wetherindegrees);
	}
}
